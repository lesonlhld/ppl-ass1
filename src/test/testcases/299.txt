Function: fibo
        Parameter: n
        Body:
            Var: n, t1 = 0, t2 = 1, nextTerm = 0;
            print("Enter the number of terms: ");
            getline(n);
            print("Fibonacci Series: ");
            For (i = 1, i <= n, i=i+1) Do
                If(i == 1) Then(
                print(" " + t1);
                Continue;
                EndIf.
            )
            If(i == 2) Then
                cout << t2 << " ";
        Continue;
        EndIf.
        nextTerm = t1 + t2;
        t1 = t2;
        t2 = nextTerm;
        
        print(nextTerm + " ");
    )
    EndFor.
    return 0;
    EndBody.
